.data
      # int resultado ;
      # int op1 = 100 ;
      # int op2 = -10 ;
resultado: .space 4
op1:       .word 100
op2:       .word -10

.text
main: 
      # resultado = op1 + op2 ;
      lw $t1 op1
      lw $t2 op2
      add $t3 $t1 $t2
      la $t4 resultado
      sw $t3 ($t4)
